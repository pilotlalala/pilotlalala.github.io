xml配置 自动映射的原理

接口IuserDao findAll方法 返回List<User>

然后IUserDaoMapper.xml 文件

显示配置 是哪个接口的映射文件 

<mapper namespace="cn.gok.IUserDao">

当执行IUserDao的时候 使用这个配置文件

当执行接口内方法 对应就是 id=findAll 方法名字 findAll

​ id = getAll 方法名字getAll 之类的

执行了id=findAll对应的方法的sql语句

<!--    findAll 就是对应接口的方法-->
<!--    使用findAll 方法 执行sql查询语句 把结果映射到resultMapUser 内的 User-->
    <select id="findAll" resultMap="resultMapUser">
        select * from bas_user
    </select>

执行完之后 需要把查询结果做自动映射

<!--   数据库的字段 映射到对应的类-->
<!--    type 就是对应类的路径-->
<!--    属性就是类内的属性-->
    <resultMap id="resultMapUser" type="cn.gok.User">
        <id column="user_id" property="userId"></id>
        <result column="user_name" property="userName"></result>
        <result column="birth" property="birth"></result>
        <result column="salary" property="salary"></result>
    </resultMap>

注意

要在pom.xml里面加入读取xml配置文件的代码

<!--修改pom文件中加上此resources即可解决-->
 <!--在bujild中配置resources，来防止我们资源导出失败的问题-->
 <resources>
   <resource>
     <directory>src/main/java</directory>
     <includes>
       <include>**/*.xml</include>
     </includes>
   </resource>
 </resources>

然后在Mybatis-config里面的<configuration>块的<environment>后面配置

<mappers>
    <mapper resource="cn/gok/IUserDaoMapper.xml"/> 这里xml配置映射文件的路径
</mappers>

动态SQL

相当于把sql语句改造成if-else

Mybatis:

    <update id="update" parameterType="cn.gok.Person">

        update person

--         trim可以去掉逗号
        <trim prefix="set" suffixOverrides=",">

            <if test="personName != null and personName !=''">
                name = #{personName},
            </if>
            <if test="personAge != null and personAge !=''">
                age = #{personAge}
            </if>

        </trim>

       where id=#{personId}
    </update>

然后就可以动态输入信息

根据ID查询多个数据

<select id="findAllByIds" resultMap="personMapper">
    select *
    from person
    where
        id
        in
        <foreach collection="ids" item="item" open="(" close=")" separator=",">
            #{item}
        </foreach>

</select>

注解形式

MyUserDao

public interface MyUserDao {

    @Select("select * from bas_user")
    @Results(id="resultMapUser" , value = {
           @Result(id = true,column="user_id",property="userId"),
            @Result(id = true,column="user_name",property="userName"),
            @Result(id = true,column="birth",property="birth"),
            @Result(id = true,column="salary",property="salary")
    })
    public List<User> findAll();
}

ExcelUtils

package cn.gok.utils;

import cn.gok.User;
import org.apache.ibatis.io.Resources;
import org.apache.poi.ss.usermodel.Row;
import org.apache.poi.xssf.usermodel.XSSFCell;
import org.apache.poi.xssf.usermodel.XSSFRow;
import org.apache.poi.xssf.usermodel.XSSFSheet;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

import java.io.*;
import java.util.ArrayList;
import java.util.List;

public class ExcelUtil {

    public static List<User> importXLS(){

        ArrayList<User> list = new ArrayList<>();
        try {
        //1、获取文件输入流
        InputStream inputStream = new FileInputStream("D://国科//web2101//人员信息.xlsx");
            //s1.读取配置文件
//            InputStream inputStream = Resources.getResourceAsStream("mybatis-config.xml");
            //XSSFWorkbook
        //2、获取Excel工作簿对象
            XSSFWorkbook workbook = new XSSFWorkbook(inputStream);
        //3、得到Excel工作表对象
            XSSFSheet sheetAt = workbook.getSheetAt(0);
            //4、循环读取表格数据
        for (Row row : sheetAt) {
            //首行（即表头）不读取
/*                if (row.getRowNum() == 0) {
                    continue;
                }*/
                //读取当前行中单元格数据，索引从0开始
                String name = row.getCell(0).getStringCellValue();
                int age = (int) row.getCell(1).getNumericCellValue();
                User user = new User();
                user.setName(name);
                user.setAge(Integer.valueOf(age));
                list.add(user);
            }
         //5、关闭流
            workbook.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
        return list;
    }

    //向Excel中写数据
    public static void writeExcel(List<User> list ) throws IOException {
        //1、获取文件输入流
        InputStream inputStream = new FileInputStream("D://国科//web2101//人员信息.xlsx");
        //s1.读取配置文件
//            InputStream inputStream = Resources.getResourceAsStream("mybatis-config.xml");
        //XSSFWorkbook
        //2、获取Excel工作簿对象
        XSSFWorkbook workbook = new XSSFWorkbook(inputStream);
        //3、得到Excel工作表对象
        XSSFSheet sheet = workbook.getSheetAt(0);
        XSSFRow firstRow = sheet.createRow(0);//第一行表头
        XSSFCell cells[] = new XSSFCell[3];

/*        String[] titles = new String[]{"id","name","age"};
        //循环设置表头信息
        for (int i=0;i<3;i++){
            cells[0]=firstRow.createCell(i);
            cells[0].setCellValue(titles[i]);
        }*/

        //遍历list,将数据写入Excel中
        for (int i=0;i<list.size();i++){
            XSSFRow row = sheet.createRow(i+1);
            User user = list.get(i);
            XSSFCell cell = row.createCell(0); //第一列
            cell.setCellValue(user.getName());
            cell=row.createCell(1); //第二列
            cell.setCellValue(user.getAge());
        }
        OutputStream out = null;
        try {
            out = new FileOutputStream("D://国科//web2101//人员信息.xlsx");
            workbook.write(out);
            out.close();
        } catch (Exception e){
            e.printStackTrace();
        }
    }

    public static void main(String[] args) {
        importXLS();
    }

}
